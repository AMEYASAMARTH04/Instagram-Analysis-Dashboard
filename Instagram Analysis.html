# Convert negative MSE to positive MSE
train_mean = -np.mean(train_scores, axis=1)
train_std  = np.std(train_scores, axis=1)
test_mean  = -np.mean(test_scores, axis=1)
test_std   = np.std(test_scores, axis=1)

plt.figure(figsize=(10,6))

# Training error line
plt.plot(train_sizes, train_mean, label="Training Error", linewidth=3)

# Test/Validation error line
plt.plot(train_sizes, test_mean, label="Validation Error", linewidth=3)

# Error shadow regions
plt.fill_between(train_sizes, train_mean - train_std, train_mean + train_std, alpha=0.2)
plt.fill_between(train_sizes, test_mean - test_std, test_mean + test_std, alpha=0.2)

plt.title("Learning Curve (BiasVariance Check)")
plt.xlabel("Training Set Size")
plt.ylabel("MSE")
plt.legend()
plt.grid(True)
